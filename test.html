        // Example for API Integration

        // Load services dynamically

        async function loadServices() {

            const response = await fetch('/api/services');

            const services = await response.json();

            const serviceCards = document.getElementById('serviceCards');


            services.forEach(service => {

                const card = document.createElement('div');

                card.classList.add('service-card', 'bg-white', 'rounded-xl', 'shadow-lg', 'p-6');

                card.innerHTML = `

                    <h3 class="text-xl font-bold mb-4">${service.name}</h3>

                    <p class="text-gray-600 mb-4">${service.description}</p>

                    <div class="mb-4">

                        <label class="block text-sm font-medium mb-2">Duration</label>

                        <select class="w-full p-2 border rounded-lg">

                            ${service.durations.map(duration => `

                                <option value="${duration.hours}">${duration.hours} Hour(s) ($${duration.price})</option>

                            `).join('')}

                        </select>

                    </div>

                    <button class="w-full bg-blue-600 text-white py-2 rounded-lg hover:bg-blue-700">

                        Select

                    </button>

                `;

                serviceCards.appendChild(card);

            });

        }

            loadServices();
