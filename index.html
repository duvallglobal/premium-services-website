    <!-- JavaScript -->

    <script>

        function generateCalendar() {

            const calendarDays = document.getElementById('calendarDays');

            calendarDays.innerHTML = '';

            for (let i = 1; i <= 31; i++) {

                const day = document.createElement('div');

                day.className = 'p-4 border rounded-lg text-center';

                day.textContent = i;

                day.onclick = () => alert('Selected Date: ' + i);

                calendarDays.appendChild(day);

            }

        }


        function toggleChat() {

            document.getElementById('chatWidget').classList.toggle('hidden');

        }


        function sendMessage() {

            const input = document.getElementById('messageInput');

            const message = input.value.trim();

            if (!message) return;

            addChatMessage(message, true);

            input.value = '';

            setTimeout(() => addChatMessage("Thank you for your message! We'll get back to you shortly.", false), 1000);

        }


        function addChatMessage(message, isUser) {

            const chatMessages = document.getElementById('chatMessages');

            const messageDiv = document.createElement('div');

            messageDiv.className = `chat-message mb-4 ${isUser ? 'text-right' : ''}`;

            messageDiv.innerHTML = `<div class="inline-block px-4 py-2 rounded-lg ${isUser ? 'bg-blue-600 text-white' : 'bg-gray-100 dark:bg-gray-700' }">${message}</div>`;

            chatMessages.appendChild(messageDiv);

            chatMessages.scrollTop = chatMessages.scrollHeight;

        }


        document.addEventListener('DOMContentLoaded', generateCalendar);

    </script>
